//矩阵转置
#include<iostream>

#define COL 10
#define ROW 10
using std::cout;
using std::endl;
using std::cin;

void Transpose(int arr[][COL])
{
    for (int i = 0; i < ROW; ++i)
    {
        for (int j = 0; j < COL; ++j)
        {
            if (i < j)
            {
                int temp = arr[i][j];
                arr[i][j] = arr[j][i];
                arr[j][i] = temp;
            }
        }
    }
}
void PrintArr(int arr[][COL])
{
    for (int i = 0; i < ROW; ++i)
    {
        for (int j = 0; j < COL; ++j)
        {
            cout << arr[i][j] << " ";
        }
        cout << endl;
    }
}

int main()
{
    int arr[ROW][COL], flag = 0;
    for (int i = 0; i < ROW; ++i)
    {
        for (int j = 0; j < COL; ++j)
        {
            arr[i][j] = flag;
            ++flag;
        }
    }
    cout << "矩阵转换前：" << endl;
    PrintArr(arr);
    cout << "矩阵转换后：" << endl;
    Transpose(arr);
    PrintArr(arr);
    return 0;
}

3、指针数组和数组指针

什么时候用指针数组？ 指针数组：（int *pArr[]）指针数组常常用于存储一些长度不相等的字符串数据。

什么时候用指针数组？ 1）当作数组用，用来获取或设置数组元素。 2）把一个比较长的数据结构传入函数，只需要一个指针就够了。 3）传入变量的地址，用来读取或修改变量的值，主要是为了修改用。

#include<iostream>

int main()
{
    char* p1 = "hello";//hello存放在静态存储区
    char* p2 = "world";
    char* pArr[2] = { p1, p2 };//指针数组
    for (int i = 0; i < 2; ++i)
    {
        std::cout << pArr[i];
    }

    int arr[] = { 2, 0, 1, 9 };
    int(*p)[4]; 
    //p = arr;//错误！！！p要指向的是数组首地址，是要指向一整块数组，而arr是数组第一个元素的地址
    p = &arr;//当数组一样使用
    std::cout << *p[0] << std::endl;
    return 0;
}

4、string和char*转化

#include<iostream>

using std::string;
using std::cout;
using std::endl;

int main()
{
    //string转char *
    string str = "abc";
    char arr[20];
    strcpy(arr,str.c_str());
    cout << arr << endl;

    //char *转string
    string str2;
    char arr2[] = "hello";
