//看星星、看月亮
#include<iostream>
#include<graphics.h>
#include<time.h>

#pragma comment(lib,"Winmm.lib")

void CreateMoon(int circle_left_x, int circle_left_y, int circle_r, int color);
void CreateStart(float size, int pos_x, int pos_y);
int main()
{
    initgraph(900, 500);
    int pos_moon_x = 340,pos_moon_y = 150,pos_moon_r = 50;//月亮位置坐标和半径
    srand((unsigned)time(NULL));

    mciSendString(L"open 匆匆那年.mp3 alias my_music", 0, 0, 0); 
    mciSendString(L"play my_music repeat",0,0,0);

    setbkcolor(BLACK);

    while (1)
    {
        //创造月亮
        CreateMoon(380, 150, 50, YELLOW);
        CreateMoon(pos_moon_x, pos_moon_y, pos_moon_r, BLACK);

        //创造星星
        for (int i = 0; i < 30; ++i)
        {
            CreateStart(20, rand() % 900, rand() % 500);//随机位置生成星星
            pos_moon_x += 2;
            if (pos_moon_x == 420)
                pos_moon_x = 340;
        }
        Sleep(1000);//暂停1秒钟
        cleardevice();//清屏
    }

    getchar();
    closegraph();
    return 0;
}
void CreateMoon(int circle_left_x, int circle_left_y, int circle_r,int color)
{
    setfillcolor(color);
    solidcircle(circle_left_x, circle_left_y, circle_r);
}
void CreateStart(float start_size, int start_positionX, int start_positionY)
{
    float point1_x = start_positionX + start_size / 2;    //星星五个点的坐标 12点位置顺时针数
    float point1_y = start_positionY + 0;
    float point2_x = start_positionX + start_size;
    float point2_y = start_positionY + start_size * 3 / 8;
    float point3_x = start_positionX + start_size / 3 + start_size / 2;
    float point3_y = start_positionY + start_size;
    float point4_x = start_positionX + start_size / 2 - start_size / 3;
    float point4_y = start_positionY + start_size;
    float point5_x = start_positionX + 0;
    float point5_y = start_positionY + start_size * 3 / 8;

    POINT pts[10];

    // 把五角星的是个点放进结构体数组pts里面
    pts[0].x = point1_x;
    pts[0].y = point1_y;
    pts[1].x = start_positionX + start_size * 5 / 8;
    pts[1].y = point2_y;

    pts[2].x = point2_x;
    pts[2].y = point2_y;
    pts[3].x = start_positionX + start_size * 11 / 16;
    pts[3].y = start_positionY + start_size * 5 / 8;

    pts[4].x = point3_x;
    pts[4].y = point3_y;
    pts[5].x = start_positionX + start_size / 2;
    pts[5].y = start_positionY + start_size * 3 / 4;

    pts[6].x = point4_x;
    pts[6].y = point4_y;
    pts[7].x = start_positionX + start_size * 5 / 16;
    pts[7].y = start_positionY + start_size * 5 / 8;

    pts[8].x = point5_x;
    pts[8].y = point5_y;
    pts[9].x = start_positionX + start_size * 3 / 8;
    pts[9].y = point2_y;

    setpolyfillmode(WINDING);
    setfillcolor(YELLOW);

    solidpolygon(pts, 10);

    /*line(point1_x, point1_y, point3_x, point3_y);
    line(point1_x, point1_y, point4_x, point4_y);
    line(point5_x, point5_y, point2_x, point2_y);
    line(point5_x, point5_y, point3_x, point3_y);
    line(point2_x, point2_y, point4_x, point4_y);*/
